import styled, { keyframes } from "styled-components";

import JobLayout from "./JobLayout";
import { getAuth } from "firebase/auth";

/** The JobseekerHome functional component takes in the display name of the user for a functional display. It also takes in the user information
 * in order to perform a matching of a user to recommended jobs. These jobs are then displayed in the JobLayout component which also includes the cover letter 
 * functionality.
 */
const JobseekerHome = ({name, user}) => {
  const [shownChars, setShownChars] = useState(0)
  const [formStatus, setFormStatus] = useState(null)

  useEffect(() => {
    const interval = setInterval(() => {
      setShownChars(c => c + 1)
    }, 30); //should be 32
    return () => clearInterval(interval);
  }, []);

  useEffect(() => { // Smoother transition with form scrolling
    window.scrollTo(0, 0)
  }, [formStatus])

  const HeroPlaceholder = name + ", we found some great opportunities just for you."

  return (
    <HeroWrapper>
      <Hero>
        <h1> {HeroPlaceholder.substring(0, shownChars)} </h1>
        {shownChars >= HeroPlaceholder.length + 10 && (
          <div>
            <ButtonWrapper>
              <Button onClick={() => { }}>Show me more</Button>
              {/* <Button onClick={() => {}}>Polish resume</Button>
                          <Button onClick={() => {}}>Write cover letter</Button> */}
            </ButtonWrapper>
            <JobLayout user={user}/>
          </div>
        )}
      </Hero>
    </HeroWrapper >
  )
}
